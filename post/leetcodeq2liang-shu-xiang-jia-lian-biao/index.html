public ListNode twoSum(ListNode l1, ListNode l2) {
        ListNode dh = new ListNode(0);
        ListNode a = l1, b = l2, pre = dh;
        int carry = 0;
        while (a != null || b != null) {
            int x = (a != null) ? a.val : 0;
            int y = (b != null) ? b.val : 0;
            int sum = carry + x + y;

            carry = sum / 10;
            pre.next = new ListNode(sum % 10);
            pre = pre.next;
            if (a != null)
                a = a.next;
            if (b != null)
                b = b.next;

        }
