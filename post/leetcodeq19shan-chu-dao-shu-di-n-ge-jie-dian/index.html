public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dh = new ListNode(0);
        dh.next = head;
        ListNode start = dh;
        ListNode end = dh;

        for (int i = 1; i <= n + 1; i++) {
            start = start.next;
        }
        while (start != null) {
            start = start.next;
            end = end.next;
        }
        end.next = end.next.next;
        return dh.next;
    }
